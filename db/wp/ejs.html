javascript 套件的用法 -- ejs 樣板引擎

專案下載： ejs.zip

程式

檔案：ejstest.js

var fs = require('fs');
var ejs = require('ejs');

var template = fs.readFileSync('book.ejs', 'utf8');

var html = ejs.render(template, {
    title: '天國來的鈔票', 
    chapter:['國王', '皇后', '大臣', '革命', '銀行家']}
);

console.log(html);

樣板

檔案：book.ejs

<h1><%= title %></h1>
<ol>
    <% for(var i=0; i<chapter.length; i++) { %>
        <li><%= chapter[i] %></li>
    <% } %>
</ol>
