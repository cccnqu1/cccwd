變數與型態

範例：基本型態

class Var1 {
    public static void main(String args[]) {
        int x = 3;
        double y = 5.9;
        char c = 'a';
        String s = "def";

        System.out.println("x="+x);
        System.out.println("y="+y);
        System.out.println("c="+c);
        System.out.println("s="+s);
    }
}
