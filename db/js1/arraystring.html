函數操作

> a = [1,2,3]
[ 1, 2, 3 ]
> a.push(4)
4
> a
[ 1, 2, 3, 4 ]
> a.push(5)
5
> a
[ 1, 2, 3, 4, 5 ]
> a.concat([6,7,8])
[ 1, 2, 3, 4, 5, 6, 7, 8 ]
> a
[ 1, 2, 3, 4, 5 ]
> b=a.concat([6,7,8])
[ 1, 2, 3, 4, 5, 6, 7, 8 ]
> b
[ 1, 2, 3, 4, 5, 6, 7, 8 ]
> a
[ 1, 2, 3, 4, 5 ]
> a.join()
'1,2,3,4,5'
> a.join(':')
'1:2:3:4:5'
> a.pop
[Function: pop]
> a.pop()
5
> a
[ 1, 2, 3, 4 ]
> a.reverse()
[ 4, 3, 2, 1 ]
> a
[ 4, 3, 2, 1 ]
> a.shift()
4
> a
[ 3, 2, 1 ]
> b
[ 1, 2, 3, 4, 5, 6, 7, 8 ]
> b.slice(2,4)
[ 3, 4 ]
> b.slice(2,6)
[ 3, 4, 5, 6 ]
> a
[ 3, 2, 1 ]
> a.sort()
[ 1, 2, 3 ]
> a
[ 1, 2, 3 ]
> b
[ 1, 2, 3, 4, 5, 6, 7, 8 ]

字串函數

> x = "0123456789"
'0123456789'
> x.substr(3,2)

> x.substr(4,5)
'45678'
> x.substring(3,7)
'3456'
> x.substring(3,9)
'345678'
> x.substring(3,10)
'3456789'
> x.length
10
