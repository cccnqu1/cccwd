英翻中系統

程式： mt.js

var log = console.log;

var e2c = { dog:"狗", cat:"貓", a: "一隻", chase:"追", eat:"吃" };

function mt(e) {
  var c = [];
  for (i in e) {
    var eword = e[i];
    var cword = e2c[eword];
    c.push(cword);
  }
  return c;
}

var c = mt(process.argv.slice(2)); // 從 process.argv 這個陣列，例如：node mt a dog chase a cat 中取出 a dog chase a cat 這個子陣列出來。
log(c);
